# Default

<Preview>
  <Story name="Defaults">
    {{
      props: {
        disabled: {
          default: boolean("Disabled", false)
        }
      },
      data() {
        return { inputValue: "" };
      },
      computed: {
        variant() {
          return this.disabled ? "disabled" : this.errored ? "error" : "";
        }
      },
      components: {
        ChecFormField,
        TextField
      },
      methods: {
        handleClick(e, value) {
          return action(
            `Text field action button was clicked with: ${value}`
          )();
        }
      },
      template: `
        <div class="py-16 flex justify-center">
          <ChecFormField
            @action-click="handleClick"
            class="w-full max-w-sm"
          >
            <TextField
              name="input-name"
              placeholder="Placeholder"
              v-model="inputValue"
              style="border: 1px solid #E5E5E5;border-radius: 8px;"
              />
          </ChecFormField>
        </div>`
    }}
  </Story>
</Preview>
